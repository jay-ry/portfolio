# Cinematic Parallax Scroll Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convert the portfolio into a full cinematic scroll experience where all four sections pin and animate in/out via GSAP ScrollTrigger with glitch/scanline wipe transitions between sections.

**Architecture:** Each section is pinned for ~2 viewport heights of scroll using GSAP ScrollTrigger `pin: true`. Content animates in on `onEnter` and out via a shared `GlitchOverlay` on `onLeave`. Lenis provides smooth scrolling and integrates with ScrollTrigger via a gsap ticker listener. A single orchestrator `useEffect` in `page.tsx` owns all scroll logic; section components expose DOM refs via `useImperativeHandle` and contain no scroll logic of their own.

**Tech Stack:** GSAP + ScrollTrigger (already installed), Lenis (already installed), React `forwardRef` + `useImperativeHandle`, CSS clip-path animation.

```tsx
"use client";
import { useEffect, useRef } from "react";
import dynamic from "next/dynamic";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import Lenis from "lenis";
import Nav from "@/components/Nav";
import Hero, { type HeroHandle } from "@/components/Hero";
import Projects, { type ProjectsHandle } from "@/components/Projects";
import AISection, { type AISectionHandle } from "@/components/AISection";
import Contact, { type ContactHandle } from "@/components/Contact";
import Cursor from "@/components/Cursor";
import GlitchOverlay from "@/components/GlitchOverlay";

gsap.registerPlugin(ScrollTrigger);

const Scene = dynamic(() => import("@/components/Scene"), { ssr: false });

export default function Home() {
  const heroRef = useRef<HeroHandle>(null);
  const projectsRef = useRef<ProjectsHandle>(null);
  const aiRef = useRef<AISectionHandle>(null);
  const contactRef = useRef<ContactHandle>(null);
  const overlayRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const lenis = new Lenis();
    lenis.on("scroll", ScrollTrigger.update);
    const tickerFn = (time: number) => { lenis.raf(time * 1000); };
    gsap.ticker.add(tickerFn);
    gsap.ticker.lagSmoothing(0);

    // Glitch wipe: clip-path scanline sweep with noise bar flash
    const glitchWipe = (onMidpoint?: () => void) => {
      const overlay = overlayRef.current!;
      const bars = Array.from(overlay.querySelectorAll<HTMLElement>(".glitch-overlay__bar"));
      gsap.timeline()
        .set(overlay, { clipPath: "inset(0 0 100% 0)", opacity: 1 })
        .to(overlay, { clipPath: "inset(0 0 0% 0)", duration: 0.25, ease: "power2.in" })
        .to(bars, { opacity: 1, x: "random(-8, 8)", duration: 0.08, stagger: 0.04 }, "<0.1")
        .call(() => onMidpoint?.())
        .to(bars, { opacity: 0, x: 0, duration: 0.08 }, ">")
        .to(overlay, { clipPath: "inset(100% 0 0 0)", duration: 0.25, ease: "power2.out" })
        .set(overlay, { opacity: 0 });
    };

    // --- Hero ---
    const hero = heroRef.current!;
    gsap.set([hero.bootLines, hero.name, hero.sub, hero.cta], { opacity: 0, y: 40 });

    ScrollTrigger.create({
      trigger: hero.section,
      start: "top top",
      end: "+=200%",
      pin: true,
      onEnter: () => {
        gsap.to([hero.bootLines, hero.name, hero.sub, hero.cta], {
          opacity: 1, y: 0, duration: 0.7, stagger: 0.15, ease: "power3.out",
        });
      },
      onLeave: () => glitchWipe(),
      onEnterBack: () => {
        gsap.to([hero.bootLines, hero.name, hero.sub, hero.cta], {
          opacity: 1, y: 0, duration: 0.4, stagger: 0.08, ease: "power2.out",
        });
      },
      onLeaveBack: () => glitchWipe(),
    });

    // --- Projects ---
    const projects = projectsRef.current!;
    gsap.set([projects.titleBlock, ...projects.cards], { opacity: 0, y: 50 });

    ScrollTrigger.create({
      trigger: projects.section,
      start: "top top",
      end: "+=200%",
      pin: true,
      onEnter: () => {
        gsap.to(projects.titleBlock, { opacity: 1, y: 0, duration: 0.6, ease: "power3.out" });
        gsap.to(projects.cards, { opacity: 1, y: 0, duration: 0.6, stagger: 0.1, ease: "power3.out", delay: 0.2 });
      },
      onLeave: () => glitchWipe(),
      onEnterBack: () => {
        gsap.to([projects.titleBlock, ...projects.cards], { opacity: 1, y: 0, duration: 0.4, stagger: 0.06, ease: "power2.out" });
      },
      onLeaveBack: () => glitchWipe(),
    });

    // --- AI Section ---
    const ai = aiRef.current!;
    gsap.set(ai.titleBlock, { opacity: 0, y: 40 });
    gsap.set(ai.terminal, { opacity: 0, y: -40 });

    ScrollTrigger.create({
      trigger: ai.section,
      start: "top top",
      end: "+=200%",
      pin: true,
      onEnter: () => {
        gsap.to(ai.titleBlock, { opacity: 1, y: 0, duration: 0.6, ease: "power3.out" });
        gsap.to(ai.terminal, { opacity: 1, y: 0, duration: 0.7, ease: "power3.out", delay: 0.2 });
      },
      onLeave: () => glitchWipe(),
      onEnterBack: () => {
        gsap.to([ai.titleBlock, ai.terminal], { opacity: 1, y: 0, duration: 0.4, stagger: 0.1, ease: "power2.out" });
      },
      onLeaveBack: () => glitchWipe(),
    });

    // --- Contact ---
    const contact = contactRef.current!;
    gsap.set([contact.titleBlock, ...contact.cards], { opacity: 0, y: 30 });

    ScrollTrigger.create({
      trigger: contact.section,
      start: "top top",
      end: "+=200%",
      pin: true,
      onEnter: () => {
        gsap.to(contact.titleBlock, { opacity: 1, y: 0, duration: 0.6, ease: "power3.out" });
        gsap.to(contact.cards, { opacity: 1, y: 0, duration: 0.5, stagger: 0.08, ease: "back.out(1.4)", delay: 0.2 });
      },
      onEnterBack: () => {
        gsap.to([contact.titleBlock, ...contact.cards], { opacity: 1, y: 0, duration: 0.4, stagger: 0.06, ease: "power2.out" });
      },
      onLeaveBack: () => glitchWipe(),
    });

    return () => {
      lenis.destroy();
      gsap.ticker.remove(tickerFn);
      ScrollTrigger.getAll().forEach(t => t.kill());
    };
  }, []);

  return (
    <>
      <Cursor />
      <Scene />
      <Nav />
      <GlitchOverlay ref={overlayRef} />
      <main className="grid-bg" style={{ position: "relative", zIndex: 10 }}>
        <Hero ref={heroRef} />
        <Projects ref={projectsRef} />
        <AISection ref={aiRef} />
        <Contact ref={contactRef} />
      </main>
    </>
  );
}
```
